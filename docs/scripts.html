def load_data() -> pd.DataFrame:
    """Load and clean one of the output files created from the update script - `for_log_tracker_...xlsx`. Warn if ticket # is missing.
    Returns
    -------
    pd.DataFrame
        df w/ 2 columns: `ticket_num` and `log_action`
    """
    latest = sorted(os.listdir(("../dat")))[-1]
    lg.info(f"Latest log tracker file: {latest}")

    df = pd.read_excel((f"../dat/{latest}"))
    df = df[["ticket_num", "death_update"]].copy()
    df.columns = ["ticket_num", "log_action"]

    m = df.ticket_num.isna()
    if len(df[m]) > 0:
        lg.warning("Missing ticket numbers in data")

    return df.copy()

    if ticket_num not in completed_list:
        res = bot.close_log(ticket_num)
        ctr = 0

        while (res != 0) and (ctr < 10):
            time.sleep(0.5)
            res = bot.close_log(ticket_num)
            ctr += 1

        if res == 0:
            completed_list.append(ticket_num)
            lg.trace(f"closed {ticket_num}")
        else:
            lg.warning(f"error closing {ticket_num}; manual check needed")

    remaining = len(logs_to_reassign) - (i + 1)

    if ticket_num not in reassigned_list:
        _ok = False
        ctr = 0

        while not _ok:
            try:
                bot.search_ticket_num(ticket_num)
                bot.reassign_log(note)
                time.sleep(1.8)
            except:
                time.sleep(1)

                ctr += 1

                    raise Exception("manual check required")
